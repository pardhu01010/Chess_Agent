import pygame
import os
import torch
import torch.nn as nn
import torch.optim as optim
import random
import numpy as np
import matplotlib.pyplot as plt
import chess

# Step 1: Define the Chess Environment (Using python-chess)
class ChessEnvironment:
    def __init__(self):
        self.board = chess.Board()
        self.game_over = False
    
    def reset(self):
        self.board = chess.Board()
        self.game_over = False
        return self.board.fen()
    
    def step(self, move):
        if move not in self.board.legal_moves:
            raise ValueError("Illegal Move")
        
        self.board.push(move)
        if self.board.is_game_over():
            self.game_over = True
            if self.board.result() == "1-0":
                return self.board.fen(), 1, True  # Win
            elif self.board.result() == "0-1":
                return self.board.fen(), -1, True  # Loss
            else:
                return self.board.fen(), 0, True  # Draw
        else:
            return self.board.fen(), 0, False  # Continue the game
    
    def get_state(self):
        return self.board.fen()
    
    def get_legal_moves(self):
        return list(self.board.legal_moves)

# Step 2: Define the Chess Model (A simple neural network for Q-learning)
class ChessModel(nn.Module):
    def __init__(self):
        super(ChessModel, self).__init__()
        self.fc1 = nn.Linear(8 * 8, 128)  # Flattened chessboard as input
        self.fc2 = nn.Linear(128, 64)
        self.fc3 = nn.Linear(64, 1)  # Output a value to represent the action value (Q-value)
    
    def forward(self, x):
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        x = self.fc3(x)
        return x

# Step 3: Define the DQN Agent with Live Graph Update
class DQNAgent:
    def __init__(self, model, optimizer, gamma=0.99, batch_size=32):
        self.model = model
        self.optimizer = optimizer
        self.gamma = gamma
        self.epsilon = 1.0  # Exploration factor
        self.epsilon_min = 0.1
        self.epsilon_decay = 0.995
        self.memory = []
        self.batch_size = batch_size
        self.episode_rewards = []  # To store rewards for each episode
        self.losses = []  # To store losses for each episode

    def get_move(self, state, legal_moves):
        if random.random() < self.epsilon:
            return random.choice(legal_moves)
        else:
            state_tensor = torch.tensor(state, dtype=torch.float32).flatten()
            q_values = self.model(state_tensor)
            best_move = legal_moves[np.argmax(q_values.detach().numpy())]  # Get the move with the highest Q-value
            return best_move
    
    def train(self):
        if len(self.memory) < self.batch_size:
            return
        batch = random.sample(self.memory, self.batch_size)
        total_loss = 0

        for state, action, reward, next_state, done in batch:
            state_tensor = torch.tensor(state, dtype=torch.float32).flatten()
            next_state_tensor = torch.tensor(next_state, dtype=torch.float32).flatten()
            target = reward + (1 - done) * self.gamma * torch.max(self.model(next_state_tensor))

            q_values = self.model(state_tensor)
            loss = torch.nn.functional.mse_loss(q_values, target)
            total_loss += loss.item()

            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()

        if self.epsilon > self.epsilon_min:
            self.epsilon *= self.epsilon_decay
        
        self.losses.append(total_loss)  # Record the loss for live plotting
        return total_loss

    def update_rewards(self, reward):
        self.episode_rewards.append(reward)
    
    def plot_metrics(self):
        plt.clf()  # Clear the previous plot
        plt.subplot(2, 1, 1)
        plt.plot(self.episode_rewards, label="Reward per Episode")
        plt.xlabel('Episode')
        plt.ylabel('Reward')
        plt.title('Reward Progress')
        
        plt.subplot(2, 1, 2)
        plt.plot(self.losses, label="Loss per Episode", color='r')
        plt.xlabel('Episode')
        plt.ylabel('Loss')
        plt.title('Loss Progress')

        plt.tight_layout()
        plt.draw()
        plt.pause(0.01)  # Pause to allow plot update

# Step 4: Define Pygame Visualization for Chess Board
class ChessGUI:
    def __init__(self, size=600):
        self.size = size
        self.screen = pygame.display.set_mode((size, size))
        pygame.display.set_caption("Chess Game")
        
        # Load the images for chess pieces
        self.piece_images = self.load_piece_images()

    def load_piece_images(self):
        # Load images for each piece type
        piece_images = {}
        pieces = ['K', 'Q', 'R', 'B', 'N', 'P']
        colors = ['w', 'b']  # white and black
        
        for color in colors:
            for piece in pieces:
                image_path = os.path.join('assets', f'{color}{piece}.png')
                if os.path.exists(image_path):
                    piece_images[f'{color}{piece}'] = pygame.image.load(image_path)
                else:
                    print(f"Image not found: {image_path}")  # Debugging message
        return piece_images

    def draw_board(self, board):
        self.screen.fill((255, 255, 255))  # White background
        square_size = self.size // 8
        
        # Draw the chessboard grid
        for row in range(8):
            for col in range(8):
                color = (255, 255, 255) if (row + col) % 2 == 0 else (0, 0, 0)
                pygame.draw.rect(self.screen, color, pygame.Rect(col * square_size, row * square_size, square_size, square_size))
        
        # Draw the chess pieces as images
        for row in range(8):
            for col in range(8):
                piece = board.piece_at(row * 8 + col)
                if piece:
                    piece_symbol = piece.symbol()
                    piece_color = 'w' if piece_symbol.isupper() else 'b'
                    piece_type = piece_symbol.upper()  # R, N, B, Q, K, P
                    
                    piece_image = self.piece_images.get(f'{piece_color}{piece_type}')
                    if piece_image:
                        piece_image = pygame.transform.scale(piece_image, (square_size, square_size))
                        self.screen.blit(piece_image, (col * square_size, row * square_size))
        
        pygame.display.flip()

# Step 5: Define the Game Loop with Multiple Episodes
def train_agent(episodes=100):
    pygame.init()
    gui = ChessGUI()
    model = ChessModel()
    optimizer = optim.Adam(model.parameters(), lr=0.001)
    agent = DQNAgent(model, optimizer)
    game_env = ChessEnvironment()

    for episode in range(episodes):
        done = False
        state = game_env.reset()
        total_episode_reward = 0
        clock = pygame.time.Clock()

        while not done:
            for event in pygame.event.get():
                if event.type == pygame.QUIT:
                    pygame.quit()
                    return
            
            legal_moves = game_env.get_legal_moves()
            if legal_moves:
                action = agent.get_move(state, legal_moves)  # Use the agent's move decision
                action_uci = action.uci()  # Convert move to UCI format
                print(f"Episode {episode + 1}, Move: {action_uci}")
                
                # Play the move
                next_state, reward, done = game_env.step(chess.Move.from_uci(action_uci))
                
                # Update rewards and plot live graph
                agent.update_rewards(reward)
                total_episode_reward += reward
                agent.train()  # Train the model after each step
                
                # Visualize the board
                gui.draw_board(game_env.board)
                
                # Control the game frame rate
                clock.tick(30)
                
                state = next_state
                if done:
                    print(f"Game Over in Episode {episode + 1}: Reward = {total_episode_reward}")
                    break
        
        # Log total reward for the episode and plot metrics after each episode
        agent.plot_metrics()

    pygame.quit()

# Initialize the plot
plt.ion()  # Turn on interactive mode for live plotting

# Run the training loop
train_agent(episodes=100)

plt.ioff()  # Turn off interactive mode

import pygame

import torch
import torch.nn as nn
import torch.optim as optim
import random
import numpy as np
import matplotlib.pyplot as plt
import chess

# Step 1: Define the Chess Environment
class ChessEnvironment:
    def __init__(self):
        self.board = chess.Board()
        self.game_over = False
    
    def reset(self):
        self.board = chess.Board()
        self.game_over = False
        return self.board.fen()
    
    def calculate_material_balance(self):
        material_values = {
            'p': 1, 'n': 3, 'b': 3, 'r': 5, 'q': 9,  # Black pieces
            'P': -1, 'N': -3, 'B': -3, 'R': -5, 'Q': -9  # White pieces
        }
        balance = 0
        for piece in str(self.board):
            balance += material_values.get(piece, 0)
        return balance
    
    def step(self, move):
        if move not in self.board.legal_moves:
            raise ValueError("Illegal Move")
        
        previous_balance = self.calculate_material_balance()
        self.board.push(move)
        new_balance = self.calculate_material_balance()
        
        # Normalize reward to the range of -1 to 1 based on material change
        reward = np.clip(new_balance - previous_balance, -1, 1)  # Reward is now between -1 and 1

        if self.board.is_game_over():
            self.game_over = True
            result = self.board.result()
            # Assign rewards for the game outcome (win: 1, draw: 0, loss: -1)
            reward += 1 if result == "1-0" else -1 if result == "0-1" else 0
            return self.board.fen(), reward, True
        else:
            return self.board.fen(), reward, False  # Continue the game

    def get_state(self):
        return self.board.fen()
    
    def get_legal_moves(self):
        return list(self.board.legal_moves)

# Helper function to convert FEN to numerical board representation
def fen_to_tensor(fen):
    piece_map = {
        'p': -1, 'n': -2, 'b': -3, 'r': -4, 'q': -5, 'k': -6,  # Black pieces
        'P': 1, 'N': 2, 'B': 3, 'R': 4, 'Q': 5, 'K': 6         # White pieces
    }
    board_tensor = np.zeros((8, 8), dtype=np.float32)

    rows = fen.split()[0].split('/')
    for i, row in enumerate(rows):
        col = 0
        for char in row:
            if char.isdigit():
                col += int(char)  # Empty squares
            else:
                board_tensor[i, col] = piece_map[char]
                col += 1
    return torch.tensor(board_tensor.flatten(), dtype=torch.float32)

# Step 2: Define the Chess Model
class ChessModel(nn.Module):
    def __init__(self, output_size=64):
        super(ChessModel, self).__init__()
        self.fc1 = nn.Linear(8 * 8, 128)
        self.fc2 = nn.Linear(128, 64)
        self.fc3 = nn.Linear(64, output_size)

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        x = self.fc3(x)
        return x

class DQNAgent:
    def __init__(self, model, optimizer, gamma=0.99, batch_size=32, memory_size=10000):
        self.model = model
        self.optimizer = optimizer
        self.gamma = gamma
        self.epsilon = 1.0
        self.epsilon_min = 0.1
        self.epsilon_decay = 0.995
        self.memory = []
        self.memory_size = memory_size
        self.batch_size = batch_size
        self.episode_rewards = []
        self.losses = []

    def get_move(self, state, legal_moves):
        move_indices = list(range(len(legal_moves)))
        
        if random.random() < self.epsilon:
            return random.choice(move_indices)
        else:
            state_tensor = fen_to_tensor(state)
            q_values = self.model(state_tensor)
            legal_q_values = [q_values[i].item() for i in move_indices]
            best_move_idx = np.argmax(legal_q_values)
            return move_indices[best_move_idx]
    
    def remember(self, state, action_idx, reward, next_state, done):
        if len(self.memory) >= self.memory_size:
            self.memory.pop(0)
        self.memory.append((state, action_idx, reward, next_state, done))

    def train(self):
        if len(self.memory) < self.batch_size:
            return

        batch = random.sample(self.memory, self.batch_size)
        total_loss = 0

        for state, action_idx, reward, next_state, done in batch:
            state_tensor = fen_to_tensor(state)
            next_state_tensor = fen_to_tensor(next_state)
            target = reward + (1 - done) * self.gamma * torch.max(self.model(next_state_tensor))

            q_values = self.model(state_tensor)
            loss = nn.functional.mse_loss(q_values[action_idx], target.unsqueeze(0))
            total_loss += loss.item()

            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()

        if self.epsilon > self.epsilon_min:
            self.epsilon *= self.epsilon_decay

        self.losses.append(total_loss)

    def update_rewards(self, reward):
        self.episode_rewards.append(reward)

    def plot_metrics(self):
        plt.clf()
        plt.subplot(2, 1, 1)
        plt.plot(self.episode_rewards, label="Reward per Episode")
        plt.xlabel('Episode')
        plt.ylabel('Reward')
        plt.title('Reward Progress')
        
        plt.subplot(2, 1, 2)
        plt.plot(self.losses, label="Loss per Episode", color='r')
        plt.xlabel('Episode')
        plt.ylabel('Loss')
        plt.title('Loss Progress')

        plt.tight_layout()
        plt.draw()

    torch.save(model.state_dict(), 'chess_model_trained.pth')

                board_tensor[i, col] = piece_map[char]
                col += 1
    return torch.tensor(board_tensor.flatten(), dtype=torch.float32)

# Step 2: Define the Chess Model
class ChessModel(nn.Module):
    def __init__(self, output_size=64):
        super(ChessModel, self).__init__()
        self.fc1 = nn.Linear(8 * 8, 128)
        self.fc2 = nn.Linear(128, 64)
        self.fc3 = nn.Linear(64, output_size)

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        x = self.fc3(x)
        return x

class DQNAgent:
    def __init__(self, model, optimizer, gamma=0.99, batch_size=32, memory_size=10000):
        self.model = model
        self.optimizer = optimizer
        self.gamma = gamma
        self.epsilon = 1.0
        self.epsilon_min = 0.1
        self.epsilon_decay = 0.995
        self.memory = []
        self.memory_size = memory_size
        self.batch_size = batch_size
        self.episode_rewards = []
        self.losses = []

    def get_move(self, state, legal_moves):
        state_tensor = fen_to_tensor(state)
        q_values = self.model(state_tensor)

        # Create a mapping of legal moves to their indices
        legal_moves_dict = {move.to_square: move for move in legal_moves}

        # Fetch Q-values for legal moves
        legal_q_values = [q_values[move.to_square].item() for move in legal_moves]
        best_move_idx = np.argmax(legal_q_values)
        best_move = legal_moves[best_move_idx]

        return best_move



    
    def remember(self, state, action_idx, reward, next_state, done):
        if len(self.memory) >= self.memory_size:
            self.memory.pop(0)
        self.memory.append((state, action_idx, reward, next_state, done))

    def train(self):
        if len(self.memory) < self.batch_size:
            return

        batch = random.sample(self.memory, self.batch_size)
        total_loss = 0

        for state, action_idx, reward, next_state, done in batch:
            state_tensor = fen_to_tensor(state)
            next_state_tensor = fen_to_tensor(next_state)
            target = reward + (1 - done) * self.gamma * torch.max(self.model(next_state_tensor))

            q_values = self.model(state_tensor)
            loss = nn.functional.mse_loss(q_values[action_idx], target.unsqueeze(0))
            total_loss += loss.item()

            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()

        if self.epsilon > self.epsilon_min:
            self.epsilon *= self.epsilon_decay

        self.losses.append(total_loss)

    def update_rewards(self, reward):
        self.episode_rewards.append(reward)

    def plot_metrics(self):
        plt.clf()
        plt.subplot(2, 1, 1)
        plt.plot(self.episode_rewards, label="Reward per Episode")
        plt.xlabel('Episode')
        plt.ylabel('Reward')
        plt.title('Reward Progress')
        
        plt.subplot(2, 1, 2)
        plt.plot(self.losses, label="Loss per Episode", color='r')
        plt.xlabel('Episode')
        plt.ylabel('Loss')
        plt.title('Loss Progress')

        plt.tight_layout()
        plt.draw()
        plt.pause(0.01)


class ChessGUI:
    def __init__(self, size=600):
        self.size = size
        self.screen = pygame.display.set_mode((size, size))
        pygame.display.set_caption("Chess Game")
        self.piece_images = self.load_piece_images()

        # Custom colors for the chessboard squares
        self.board_color_light = (222, 184, 135)  # Light brown (for white squares)
        self.board_color_dark = (139, 69, 19)    # Dark brown (for black squares)

    def load_piece_images(self):
        piece_images = {}
        pieces = ['K', 'Q', 'R', 'B', 'N', 'P']
        colors = ['w', 'b']
        
        for color in colors:
            for piece in pieces:
                image_path = os.path.join('assets', f'{color}{piece}.png')
                if os.path.exists(image_path):
                    piece_images[f'{color}{piece}'] = pygame.image.load(image_path)
                else:
                    print(f"Image not found: {image_path}")
        return piece_images

    def draw_board(self, board):
        self.screen.fill((255, 255, 255))  # Fill the screen with white background (for the border)
        square_size = self.size // 8
        
        for row in range(8):
            for col in range(8):
                # Alternate between light and dark brown for the squares
                color = self.board_color_light if (row + col) % 2 == 0 else self.board_color_dark
                pygame.draw.rect(self.screen, color, pygame.Rect(col * square_size, row * square_size, square_size, square_size))
        
        # Draw the pieces on the board
        for row in range(8):
            for col in range(8):
                piece = board.piece_at(row * 8 + col)
                if piece:
                    piece_symbol = piece.symbol()
                    piece_color = 'w' if piece_symbol.isupper() else 'b'
                    piece_type = piece_symbol.upper()
                    piece_image = self.piece_images.get(f'{piece_color}{piece_type}')
                    if piece_image:
                        piece_image = pygame.transform.scale(piece_image, (square_size, square_size))
                        self.screen.blit(piece_image, (col * square_size, row * square_size))
        
        pygame.display.flip()  # Update the screen with the new drawing

# Step 5: Define the Game Loop with Multiple Episodes
def train_agent(episodes= 5):
    pygame.init()
    gui = ChessGUI()
    model = ChessModel()
    optimizer = optim.Adam(model.parameters(), lr=0.001)
    agent = DQNAgent(model, optimizer)
    game_env = ChessEnvironment()
    max_moves = 200  # Define maximum moves per episode

    episode_rewards = []  # To store the total reward for each episode

    for episode in range(episodes):
        done = False
        state = game_env.reset()
        total_episode_reward = 0
        clock = pygame.time.Clock()
        move_count = 0  # Track moves in the episode

        while not done and move_count < max_moves:
            for event in pygame.event.get():
                if event.type == pygame.QUIT:
                    pygame.quit()
                    return

            legal_moves = game_env.get_legal_moves()
            if legal_moves:
                action_idx = agent.get_move(state, legal_moves)
                action = legal_moves[action_idx]
                
                next_state, reward, done = game_env.step(action)
                
                agent.remember(state, action_idx, reward, next_state, done)
                agent.train()
                agent.update_rewards(reward)
                total_episode_reward += reward
                gui.draw_board(game_env.board)
                clock.tick(30)

                state = next_state
                move_count += 1
                if done:
                    print(f"Game Over in Episode {episode + 1} with result: {game_env.board.result()}")

        # Store total reward for this episode
        episode_rewards.append(total_episode_reward)

        # Print the total reward for this episode
        print(f"Episode {episode + 1} - Total Reward: {total_episode_reward}")

        # If max moves are reached, penalize agent for not finishing the game
        if move_count >= max_moves:
            print("Reached max moves, penalizing agent.")
            agent.update_rewards(-0.1)  # Small penalty for unresolved game

        agent.plot_metrics()  # Plot reward and loss metrics after each episode

    # Calculate and display the final reward summary
    average_reward = sum(episode_rewards) / len(episode_rewards)
    print(f"\nFinal Reward Summary: Average Reward = {average_reward:.2f} per episode")

    # Save the trained model after completing the training
    torch.save(model.state_dict(), 'chess_model_trained.pth')
    print("Model saved successfully as 'chess_model_trained.pth'")

train_agent()
